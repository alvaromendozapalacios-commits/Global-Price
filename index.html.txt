<button onclick="calcular()">Calcular</button>

<h3 id="resultado"></h3>

<script>
function calcular() {
  let producto = document.getElementById("producto").value;
  let pais = document.getElementById("pais").value;

  let precios = {
    audifonos: { usa: 40, china: 25 },
    mouse: { usa: 20, china: 12 }
  };

  let tipoCambio = {
    peru: 3.8,
    mexico: 17
  };

  let envio = 10;
  let impuesto = 0.18;

  let resultadoTexto = "";

  for (let origen in precios[producto]) {
    let precioUSD = precios[producto][origen];
    let total = (precioUSD + envio) * tipoCambio[pais];
    total = total + (total * impuesto);

    resultadoTexto += origen.toUpperCase() + ": " + total.toFixed(2) + "<br>";
  }

  document.getElementById("resultado").innerHTML = resultadoTexto;
}
</script>
